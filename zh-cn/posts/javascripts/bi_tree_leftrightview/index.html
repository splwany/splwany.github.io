        _leftView(root.left, level + 1, viewArr);
        _leftView(root.right, level + 1, viewArr);
    };
    const _rightView = (root, level, viewArr) => {
        if (!root) return;
        if (level === viewArr.length) viewArr.push(root.value);

        _rightView(root.left, level + 1, viewArr);
    };

            return viewArr;
        },
        rightView() {

}
